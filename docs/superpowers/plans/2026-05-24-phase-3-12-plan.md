## Task 3: Create NewEmployeeNotification

**Files:**
- Create: `app/Notifications/NewEmployeeNotification.php`

- [ ] **Step 1: Create the class**

```php
<?php

namespace App\Notifications;

use App\Models\Employee;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Notifications\Messages\MailMessage;
use Illuminate\Notifications\Notification;

/**
 * Sent to all users with employees.set_credentials permission when a new
 * employee is created. subtype is 'credentials_ready' (has email, account
 * auto-created) or 'credentials_required' (no email, IT must set manually).
 */
class NewEmployeeNotification extends Notification implements ShouldQueue
{
    use Queueable;

    public function __construct(
        private readonly Employee $employee,
        private readonly string $subtype, // 'credentials_ready' | 'credentials_required'
    ) {}

    /** @return list<string> */
    public function via(object $notifiable): array
    {
        return ['database', 'mail'];
    }

    /** @return array<string, mixed> */
    public function toDatabase(object $notifiable): array
    {
        return [
            'type'             => 'new_employee',
            'subtype'          => $this->subtype,
            'employee_id'      => $this->employee->id,
            'employee_name'    => $this->employee->name,
            'employee_code'    => $this->employee->code,
            'default_password' => $this->subtype === 'credentials_ready' ? $this->employee->code : null,
        ];
    }

    public function toMail(object $notifiable): MailMessage
    {
        $isReady = $this->subtype === 'credentials_ready';

        $mail = (new MailMessage)
            ->subject('[IT Service Desk] New Employee — ' . ($isReady ? 'Account Created' : 'Action Required'))
            ->greeting('New employee added: ' . $this->employee->name . ' (' . $this->employee->code . ')');

        if ($isReady) {
            $mail->line('A system account has been created automatically.')
                 ->line('Default login password: **' . $this->employee->code . '**')
                 ->line('The employee can log in with their email address.');
        } else {
            $mail->line('This employee has no email address.')
                 ->line('Please set a Username and Password manually in the Employee Profile.');
        }

        return $mail;
    }
}
```

```php
public function credentials(Request $request, Employee $employee): JsonResponse
{
    abort_unless((bool) $request->user()?->hasPermission('employees.set_credentials'), 403);

    if (! empty($employee->email)) {
        return response()->json(['message' => 'Employee has email — account is created automatically.'], 422);
    }

    $data = $request->validate([
        // unique:users,username prevents duplicate usernames automatically.
        'username' => ['required', 'string', 'max:255', 'unique:users,username'],
        'password' => ['required', 'string', 'min:6', 'confirmed'],
    ]);

    $this->service->createUserWithCredentials($employee, $data['username'], $data['password']);

    \App\Models\AuditLog::record('Created user account', "{$employee->name} ({$employee->code})");

    return response()->json(['message' => 'success'], 201);
}
```

## Task 7: GroupRoleController — User.role sync on membership change

**Files:**
- Modify: `app/Http/Controllers/Api/GroupRoleController.php`

- [ ] **Step 1: Add a private sync helper**

Add inside `GroupRoleController`:

```php
use App\Models\Employee;
use App\Models\User;

/**
 * After employees are synced to a group, update User.role for all
 * employees whose membership changed (added or removed).
 * Pass the full set of employee IDs the group will have after the update.
 */
private function syncUserRoles(GroupRole $group, array $newEmployeeIds): void
{
    // Employees now in the group → set their User.role to the group's role.
    if ($group->role && count($newEmployeeIds) > 0) {
        $emails    = Employee::whereIn('id', $newEmployeeIds)->pluck('email')->filter();
        $usernames = Employee::whereIn('id', $newEmployeeIds)->pluck('username')->filter();
        User::where(function ($q) use ($emails, $usernames) {
            $q->whereIn('email', $emails)->orWhereIn('username', $usernames);
        })->update(['role' => $group->role]);
    }
}

/**
 * Resets User.role to 'user' for employees no longer in any group.
 */
private function resetRoleForOrphans(array $removedEmployeeIds): void
{
    foreach ($removedEmployeeIds as $empId) {
        $employee = Employee::find($empId);
        if (! $employee) {
            continue;
        }

        // If this employee is still in another group, keep that group's role.
        $otherGroup = $employee->groupRoles()->first();
        $newRole = $otherGroup?->role ?? 'user';

        if ($employee->email) {
            User::where('email', $employee->email)->update(['role' => $newRole]);
        } elseif ($employee->username) {
            User::where('username', $employee->username)->update(['role' => $newRole]);
        }
    }
}
```

```php
public function update(Request $request, GroupRole $groupRole): JsonResponse
{
    $this->gate($request);
    $data = $this->validateData($request);

    $oldEmployeeIds = $groupRole->employees->pluck('id')->all();
    $newEmployeeIds = $data['employee_ids'] ?? [];

    $groupRole->update(['name' => $data['name'], 'role' => $data['role'] ?? null]);
    $groupRole->employees()->sync($newEmployeeIds);
    $groupRole->departments()->sync($data['department_ids'] ?? []);

    // Sync roles for new members.
    $this->syncUserRoles($groupRole, $newEmployeeIds);
    // Reset role for employees removed from this group.
    $removed = array_diff($oldEmployeeIds, $newEmployeeIds);
    if (count($removed) > 0) {
        $this->resetRoleForOrphans($removed);
    }

    AuditLog::record('Updated role group', $groupRole->name);

    return response()->json(['data' => $this->present($groupRole), 'message' => 'success']);
}
```

```php
public function destroy(Request $request, GroupRole $groupRole): JsonResponse
{
    $this->gate($request);

    if ((int) AppSetting::get('default_employee_group_id', 0) === $groupRole->id) {
        AppSetting::put('default_employee_group_id', null);
    }

    $memberIds = $groupRole->employees->pluck('id')->all();
    $name = $groupRole->name;

    return response()->json(['message' => 'success']);
}
```

```php
<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use Illuminate\Http\JsonResponse;
use Illuminate\Http\Request;

/**
 * Manages the current user's database notifications (in-app bell).
 */
class NotificationController extends Controller
{
    /** Returns the 30 most recent notifications for the authenticated user. */
    public function index(Request $request): JsonResponse
    {
        $notifications = $request->user()
            ->notifications()
            ->latest()
            ->limit(30)
            ->get()
            ->map(fn ($n) => [
                'id'         => $n->id,
                'data'       => $n->data,
                'read'       => ! is_null($n->read_at),
                'created_at' => $n->created_at->diffForHumans(),
            ]);

        $unread = $request->user()->unreadNotifications()->count();

        return response()->json(['data' => $notifications, 'unread' => $unread]);
    }

    /** Marks a single notification as read. */
    public function markRead(Request $request, string $id): JsonResponse
    {
        $notification = $request->user()->notifications()->where('id', $id)->firstOrFail();
        $notification->markAsRead();

        return response()->json(['message' => 'success']);
    }

    /** Marks all notifications for the current user as read. */
    public function markAllRead(Request $request): JsonResponse
    {
        $request->user()->unreadNotifications->markAsRead();

        return response()->json(['message' => 'success']);
    }
}
```
